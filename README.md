      Object result;
      try{
          result=engine.eval(ans);
          jTextField1.setText(result.toString());
      }catch(ScriptException ex){
          Logger.getLogger(NewJFrame3.class.getName()).log(Level.SEVERE,null,ex);    
      }
    }
